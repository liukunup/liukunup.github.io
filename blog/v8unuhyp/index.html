Docker

约 171 字小于 1 分钟

2025-10-31

常见问题

  • 容器配置/etc/hosts
Docker
docker run \
  --add-host subdomain.yourdomain.lan:192.168.100.88 \
  hello-world
Docker Compose
services:
  myapp:
    image: hello-world
    extra_hosts:
      - "subdomain.yourdomain.lan:192.168.100.88"
  • 容器配置 DNS 服务器
Docker
docker run \
  --dns 114.114.114.114 \
  --dns 8.8.8.8 \
  hello-world
Docker Compose
services:
  myapp:
    image: hello-world
    dns:
      - 114.114.114.114
      - 8.8.8.8
    dns_search:
      - subdomain.yourdomain.lan
  • 容器配置证书
镜像构建时(推荐)
# 复制证书
COPY ssl/certs/ca.crt /usr/local/share/ca-certificates/
# 更新证书
RUN update-ca-certificates
挂载主机目录(共享)
services:
  myapp:
    image: hello-world
    volumes:
      - /etc/ssl/certs:/etc/ssl/certs:ro
直接挂载证书

通常情况下都有效，但是不推荐！

services:
  myapp:
    image: hello-world
    volumes:
      - ./certs/server.crt:/etc/ssl/certs/server.crt:ro
      - ./certs/server.key:/etc/ssl/private/server.key:ro
      - ./certs/ca.crt:/etc/ssl/certs/ca.crt:ro
    environment:
      - SSL_CERT_FILE=/etc/ssl/certs/server.crt
      - SSL_KEY_FILE=/etc/ssl/private/server.key
